```javaclass Solution {
    public int[] getMaximumXor(int[] nums, int maximumBit) {
        int n = nums.length;
        int res[] = new int[n];
        int k = (int)(Math.pow(2,maximumBit)-1);
        res[0] = nums[0];

        for(int i=1;i<n;i++){
            nums[i] ^= nums[i-1];
        }

        for(int i=0;i<nums.length;i++){
            res[n-i-1] = nums[i] ^ k;
        }
        return res;
    }
}
```
